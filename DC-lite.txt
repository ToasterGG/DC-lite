let messageTimestamps = [];
let isMuted = false;
let muteExpiresAt = null;

// ─────────────────────────────────────────────
//  localStorage helpers
// ─────────────────────────────────────────────
function lsGet(k)    { try { return localStorage.getItem(k); }  catch(_) { return null; } }
function lsSet(k, v) { try { localStorage.setItem(k, v); }      catch(_) {} }

// ─────────────────────────────────────────────
//  Rate Limiting Functions
// ─────────────────────────────────────────────
function isUserMuted() {
  if (!isMuted) return false;
  if (Date.now() > muteExpiresAt) {
    isMuted = false;
    muteExpiresAt = null;
    document.getElementById("message-input").disabled = false;
    document.getElementById("send-button").disabled = false;
    document.getElementById("upload-button").disabled = false;
    hideRateLimitWarning();
    return false;
  }
  return true;
}

function checkMessageRate() {
  const now = Date.now();
  const cutoff = now - RATE_LIMIT_WINDOW;
  
  messageTimestamps = messageTimestamps.filter(ts => ts > cutoff);
  const messageCount = messageTimestamps.length;
  
  if (isUserMuted()) {
    showRateLimitWarning("🚫 You are muted for " + Math.ceil((muteExpiresAt - now) / 1000) + "s");
    return false;
  }
  
  if (messageCount >= MUTE_THRESHOLD) {
    isMuted = true;
    muteExpiresAt = now + MUTE_DURATION;
    document.getElementById("message-input").disabled = true;
    document.getElementById("send-button").disabled = true;
    document.getElementById("upload-button").disabled = true;
    showRateLimitWarning("🚫 MUTED for 5 minutes!");
    setTimeout(() => { isUserMuted(); }, MUTE_DURATION);
    return false;
  }
  
  if (messageCount >= WARNING_THRESHOLD) {
    showRateLimitWarning("⚠️ Slow down! " + messageCount + "/6 before mute");
    return false;
  }
  
  if (messageCount >= MESSAGE_RATE_LIMIT) {
    showRateLimitWarning("⚠️ Rate limited: " + messageCount + " messages");
    return false;
  }
  
  hideRateLimitWarning();
  return true;
}

function hashString(str) {
  let h = 2166136261;
  for (let i = 0; i < str.length; i++) {
    h ^= str.charCodeAt(i);
    h = (h * 16777619) >>> 0;
  }
  return h;
}

function mulberry32(seed) {
  return function() {
    seed |= 0; seed = seed + 0x6D2B79F5 | 0;
    let t = Math.imul(seed ^ seed >>> 15, 1 | seed);
    t = t + Math.imul(t ^ t >>> 7, 61 | t) ^ t;
    return ((t ^ t >>> 14) >>> 0) / 4294967296;
  };
}

function deriveIdFromCredentials(username, password) {
  const seed = hashString(username + ":" + password);
  const rand  = mulberry32(seed);
  let id = String(Math.floor(rand() * 9) + 1);
  for (let i = 0; i < 14; i++) id += String(Math.floor(rand() * 10));
  return id;
}

function idKey(username) { return "dl_id_" + username; }

function checkDuplicateUsername(username, callback) {
  db.ref("profiles").once("value", snap => {
    const profiles = snap.val() || {};
    let count = 0;
    let duplicateUsers = [];
    Object.keys(profiles).forEach(u => {
      if (u === username) {
        count++;
        duplicateUsers.push(u);
      }
    });
    callback(count > 1 ? duplicateUsers : null);
  });
}

function checkDuplicateId(id, callback) {
  db.ref("profiles").once("value", snap => {
    const profiles = snap.val() || {};
    let count = 0;
    let duplicateUsers = [];
    Object.entries(profiles).forEach(([username, data]) => {
      if (data.id === id) {
        count++;
        duplicateUsers.push(username);
      }
    });
    callback(count > 1 ? duplicateUsers : null);
  });
}

function loadOrCreateId(username, password) {
  const derivedId = deriveIdFromCredentials(username, password);
  
  // Check for duplicate ID
  checkDuplicateId(derivedId, duplicates => {
    if (duplicates) {
      console.log("Duplicate ID detected for:", duplicates);
      alert("Account error: Duplicate ID detected. Please contact admin.");
      return;
    }
    
    db.ref("profiles/" + username + "/id").set(derivedId, () => {
      lsSet(idKey(username), derivedId);
      currentUserId = derivedId;
      showIdInSettings(derivedId);
    });
  });
}

function showIdInSettings(id) {
  const section = document.getElementById("settings-id-section");
  const display = document.getElementById("settings-id-display");
  if (!section || !display) return;
  section.style.display = "block";
  display.textContent = id;
}

const userIdCache = {};

function fetchUserId(username, callback) {
  if (userIdCache[username]) { callback(userIdCache[username]); return; }
  db.ref("profiles/" + username + "/id").once("value", snap => {
    const id = snap.val() || "ID unavailable";
    userIdCache[username] = id;
    callback(id);
  });
}

// ─────────────────────────────────────────────
//  Device fingerprint
// ─────────────────────────────────────────────
function getDeviceFingerprint() {
  let fp = lsGet("dl_device_fp");
  if (fp) return fp;
  const traits = [
    navigator.userAgent,
    navigator.language,
    screen.width + "x" + screen.height,
    new Date().getTimezoneOffset(),
    Math.random().toString(36).slice(2)
  ].join("|");
  let h = 2166136261;
  for (let i = 0; i < traits.length; i++) {
    h ^= traits.charCodeAt(i);
    h = (h * 16777619) >>> 0;
  }
  fp = "fp_" + h.toString(16).padStart(8, "0");
  lsSet("dl_device_fp", fp);
  return fp;
}

function checkDeviceBlocked(fp, callback) {
  db.ref("blocked/" + fp).once("value", snap => {
    callback(!!snap.val());
  });
}

function asbEmergencyClear() {
  if (!confirm("EMERGENCY CLEAR ALL DATA (except Admin)?\n\nThis is IRREVERSIBLE.")) return;

  db.ref("profiles/Admin").once("value", snap => {
    const adminProfile = snap.val();
    const wipeNodes = ["messages", "presence", "dmInvites", "blocked", "banned", "muted"];
    const deletes = wipeNodes.map(node => db.ref(node).set(null));

    db.ref("profiles").once("value", profilesSnap => {
      const profiles = profilesSnap.val() || {};
      const updates = {};
      Object.keys(profiles).forEach(u => {
        if (u !== "Admin") updates[u] = null;

      if (adminProfile) db.ref("profiles/Admin").set(adminProfile);
      alert("Done. Database cleared (Admin preserved).");
    });
  });
  closeAdminPopupMenu();
}

function asbDeleteOldMessages() {
  const cutoff = Date.now() - (60 * 60 * 1000);
  db.ref("messages").once("value", snap => {
    const data = snap.val();
    if (!data) { alert("No messages."); return; }
    let deleted = 0;
    const promises = [];
    Object.entries(data).forEach(([ch, msgs]) => {
      if (!msgs) return;
      const updates = {};
      Object.entries(msgs).forEach(([key, msg]) => {
        if ((msg.ts || 0) < cutoff) { updates[key] = null; deleted++; }
      });
      if (Object.keys(updates).length) promises.push(db.ref("messages/" + ch).update(updates));
    });
    Promise.all(promises).then(() => alert("Deleted " + deleted + " messages >1h old."));
  });
  closeAdminPopupMenu();
}

function asbCheckDuplicates() {
  db.ref("profiles").once("value", snap => {
    const profiles = snap.val() || {};
    const usernames = Object.keys(profiles);
    const ids = {};
    let duplicates = [];
    
    // Check for duplicate usernames
    const usernameCount = {};
    usernames.forEach(u => {
      usernameCount[u] = (usernameCount[u] || 0) + 1;
    });
    Object.entries(usernameCount).forEach(([u, count]) => {
      if (count > 1) {
        duplicates.push("Username '" + u + "' appears " + count + " times - DELETING BOTH");
        db.ref("profiles/" + u).remove();
      }
    });
    
    // Check for duplicate IDs
    usernames.forEach(u => {
      const id = profiles[u].id;
      if (!id) return;
      if (ids[id]) {
        duplicates.push("ID '" + id + "' shared by " + ids[id] + " & " + u);
      } else {
        ids[id] = u;
      }
    });
    
    if (duplicates.length === 0) {
      alert("✅ No duplicates found!");
    } else {
      alert("⚠️ Duplicates found:\n" + duplicates.join("\n"));
    }
  });
  closeAdminPopupMenu();
}

function login() {
  const username = document.getElementById("login-username").value.trim();
  const password = document.getElementById("login-password").value.trim();

  if (!username || !password) {
    alert("Username and password required.");
    return;
  }

  // Check for duplicate username at login
  checkDuplicateUsername(username, duplicates => {
    if (duplicates && duplicates.length > 1) {
      alert("ERROR: Duplicate usernames detected. Both accounts deleted.");
      duplicates.forEach(u => db.ref("profiles/" + u).remove());
      return;
    }

    checkDeviceBlocked(getDeviceFingerprint(), blocked => {
      if (blocked) {
        alert("Device blocked.");
        return;
      }

      db.ref("banned/" + username).once("value", snap => {
        if (snap.val()) {
          alert("Account banned.");
          return;
        }

        currentUser = username;
        currentPassword = password;
        loadOrCreateId(username, password);

        document.getElementById("login-overlay").style.display = "none";
        document.getElementById("user-name").textContent = username;
        document.getElementById("user-pfp").src = lsGet("dl_pfp_" + username) || "";

        if (username === ADMIN_USER && password === ADMIN_PASS) {
          isAdmin = true;
          document.getElementById("settings-icon").textContent = "⚙⚡";
          document.getElementById("settings-icon").removeEventListener("click", openSettings);
          document.getElementById("settings-icon").addEventListener("click", (e) => {
            e.stopPropagation();
            toggleAdminPopupMenu();
          });
        }

        loadMessages(currentChannel);
        watchPresence();
        watchFriends();
        loadFriendsList();
      });
    });
  });
}

function sendMessage() {
  const input = document.getElementById("message-input");
  let text = input.value.trim();

  if (!text && !currentVideoData) return;

  const msg = {
    user: currentUser,
    type: currentVideoData ? "video" : "text",
    ts: Date.now()
  };

  if (currentVideoData) {
    msg.videoData = currentVideoData;
    msg.videoFilename = currentVideoFilename;
    currentVideoData = null;
    currentVideoFilename = null;
  } else {
    msg.text = text;
  }

  db.ref("messages/" + currentChannel).push(msg);
  input.value = "";
  document.getElementById("messages").scrollTop = document.getElementById("messages").scrollHeight;
}

function loadMessages(channel) {
  currentChannel = channel;
  const messagesDiv = document.getElementById("messages");
  messagesDiv.innerHTML = "";
  document.getElementById("header").textContent = "#" + channel;

  db.ref("messages/" + channel).on("child_added", snap => {
    const msg = snap.val();
    const key = snap.key;
    displayMessage(msg, key, channel);
  });
}

function displayMessage(msg, key, channel) {
  const messagesDiv = document.getElementById("messages");
  const row = document.createElement("div");
  row.className = "message-row";

  const pfp = document.createElement("img");
  pfp.className = "message-pfp";
  pfp.src = lsGet("dl_pfp_" + msg.user) || "";

  const content = document.createElement("div");
  content.className = "message-content";

  const username = document.createElement("span");
  username.className = "message-username has-id-tooltip";
  username.textContent = msg.user;

  username.addEventListener("mouseenter", () => {
    fetchUserId(msg.user, id => {
      let tooltip = username.querySelector(".id-tooltip");
      if (!tooltip) {
        tooltip = document.createElement("span");
        tooltip.className = "id-tooltip";
        username.appendChild(tooltip);
      }
      tooltip.textContent = id;
    });
  });

  content.appendChild(username);

  if (msg.type === "text") {
    const text = document.createElement("span");
    text.className = "message-text";
    text.innerHTML = msg.text.replace(/@(\S+)/g, '<span class="mention-highlight">@$1</span>');
    content.appendChild(text);

    if (msg.text.includes("@" + currentUser)) {
      row.classList.add("mentioned");
    }
  } else if (msg.type === "video") {
    const videoWrapper = document.createElement("div");
    videoWrapper.className = "message-video-thumb-wrapper";

    const video = document.createElement("video");
    video.src = msg.videoData;
    video.controls = false;

    const thumb = document.createElement("canvas");
    thumb.className = "message-video-thumb";
    thumb.width = 160;
    thumb.height = 90;

    const ctx = thumb.getContext("2d");
    video.addEventListener("loadedmetadata", () => {
      video.currentTime = video.duration / 2;
    });
    video.addEventListener("seeked", () => {
      ctx.drawImage(video, 0, 0, 160, 90);
    });

    const play = document.createElement("div");
    play.className = "message-video-play";
    play.textContent = "▶";

    videoWrapper.appendChild(thumb);
    videoWrapper.appendChild(play);
    videoWrapper.addEventListener("click", () => openVideoModal(msg.videoData, msg.videoFilename));

    content.appendChild(videoWrapper);
  }

  row.appendChild(pfp);
  row.appendChild(content);
  messagesDiv.appendChild(row);
  messagesDiv.scrollTop = messagesDiv.scrollHeight;
}

function saveSettings() {

  const dmPref = document.getElementById("dm-pref").value;

  if (pfpInput.files.length > 0) {

    reader.onload = (e) => {
      const pfpData = e.target.result;
      lsSet("dl_pfp_" + currentUser, pfpData);
      document.getElementById("user-pfp").src = pfpData;

    document.body.style.backgroundColor = bgColor;
    lsSet("dl_bg_" + currentUser, bgColor);
  }

  db.ref("profiles/" + currentUser + "/dmPref").set(dmPref);
  alert("Settings saved!");
  closeSettings();
}

function handleFileUpload(e) {
  const files = e.target.files;
  for (let file of files) {
    const reader = new FileReader();
    reader.onload = (evt) => {
      const data = evt.target.result;
      if (file.type.startsWith("video/")) {
        currentVideoData = data;
        currentVideoFilename = file.name;
        alert("Video: " + file.name + "\nClick Send to upload.");
      } else if (file.type.startsWith("image/")) {
        const msg = {
          user: currentUser,
          type: "image",
          imageData: data,
          ts: Date.now()
        };
        db.ref("messages/" + currentChannel).push(msg);
      } else {
        const msg = {
          user: currentUser,
          type: "file",
          fileName: file.name,
          fileSize: file.size,
          fileData: data,
          ts: Date.now()
        };
        db.ref("messages/" + currentChannel).push(msg);
      }
    };
    reader.readAsDataURL(file);
  }
  e.target.value = "";
}

function loadFriendsList() {
  const list = document.getElementById("friends-list");
  list.innerHTML = "";

  db.ref("dmInvites/" + currentUser).on("child_added", snap => {
    const from = snap.key;
    addFriendToList(from, list);
  });
}

function addFriendToList(friend, list) {
  const item = document.createElement("div");
  item.className = "friend";
  item.innerHTML = '<span class="friend-label">' + friend + '</span>';

  const deleteBtn = document.createElement("button");
  deleteBtn.className = "friend-delete";
  deleteBtn.textContent = "✕";
  deleteBtn.addEventListener("click", () => {
    db.ref("dmInvites/" + currentUser + "/" + friend).remove();
    item.remove();
  });

  item.appendChild(deleteBtn);
  item.addEventListener("click", () => {
    openDM(friend);
  });

  list.appendChild(item);
}

function openDM(friend) {
  currentChannel = "dm_" + [currentUser, friend].sort().join("_");
  document.getElementById("header").textContent = "@" + friend;
  loadMessages(currentChannel);
}

document.getElementById("add-friend-button").addEventListener("click", () => {
  const input = document.getElementById("add-friend-input");
  const friend = input.value.trim();

  if (!friend) return;

  db.ref("dmInvites/" + friend + "/" + currentUser).set({ ts: Date.now() });
  input.value = "";
  alert("DM request sent to " + friend);
});
