async function hashPassword(pw) {
  try {
    if (window.crypto && window.crypto.subtle) {
      const buf = await crypto.subtle.digest("SHA-256", new TextEncoder().encode(pw));
      return Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2, "0")).join("");
    }
  } catch (e) {}
  if (window.CryptoJS && CryptoJS.SHA256) {
    return CryptoJS.SHA256(pw).toString(CryptoJS.enc.Hex);
  }
  let hash = 5381;
  for (let i = 0; i < pw.length; i++) {
    hash = ((hash << 5) + hash) + pw.charCodeAt(i);
    hash |= 0;
  }
  return ("00000000" + (hash >>> 0).toString(16)).slice(-8);
}

async function createAccount(u, p) {
  u = normalizeUsername(u);
  const hash = await hashPassword(p);
  try {
    const userRef = db.ref("users/" + u);
    const existing = await userRef.once("value");
    if (existing.val()) {
      showError("Username already taken");
      return false;
    }
    await userRef.set({ passwordHash: hash, createdAt: Date.now() });
  } catch (e) {}
  try {
    const users = JSON.parse(localStorage.getItem("dl_users") || "{}");
    if (users[u]) {
      showError("Username already taken");
      return false;
    }
    users[u] = { passwordHash: hash, createdAt: Date.now() };
    localStorage.setItem("dl_users", JSON.stringify(users));
  } catch (e) {}
  return true;
}

async function signIn(u, p) {
  u = normalizeUsername(u);
  const hash = await hashPassword(p);
  try {
    const userRef = db.ref("users/" + u);
    const snap = await userRef.once("value");
    const data = snap.val();
    if (!data) {
      throw new Error("not_found");
    }
    const stored = data.passwordHash;
    if (stored && typeof stored === "string") {
      if (stored !== hash) {
        showError("Incorrect password");
        return false;
      }
      return true;
    }
    if (data.password && data.password === p) {
      await userRef.set({ passwordHash: hash, createdAt: data.createdAt || Date.now() });
      return true;
    }
    showError("Incorrect password");
    return false;
  } catch (e) {
    try {
      const users = JSON.parse(localStorage.getItem("dl_users") || "{}");
      const data = users[u];
      if (!data) {
        showError("Account not found");
        return false;
      }
      if (data.passwordHash !== hash) {
        showError("Incorrect password");
        return false;
      }
      return true;
    } catch (e2) {
      showError("Unable to sign in");
      return false;
    }
  }
}

  }

  currentUser = u;

  document.getElementById("login-overlay").style.display = "none";
  applyUserToUI();
  applySettings();
  restoreState();
  const last = localStorage.getItem("dl_lastChannel") || "general";
  setActiveChannel(last);
  loadMessages(currentChannel);
}

// SETTINGS
function applySettings() {
  const bg = localStorage.getItem("dl_bgColor");
  const dmPref = localStorage.getItem("dl_dmPref");
  const pfp = localStorage.getItem("dl_pfp");

  if (bg) {
    document.body.style.backgroundColor = bg;
    document.getElementById("main").style.backgroundColor = bg;
    document.getElementById("sidebar").style.backgroundColor = bg;
    document.getElementById("sidebar-inner").style.backgroundColor = bg;
    document.getElementById("bg-picker").value = bg;
  }

  if (dmPref) document.getElementById("dm-pref").value = dmPref;

  if (pfp) currentPfp = pfp;
  if (currentUser) applyUserToUI();
}

function saveSettings() {
  const bg = document.getElementById("bg-picker").value;
  const dmPref = document.getElementById("dm-pref").value;
  const pfpFile = document.getElementById("pfp-input").files[0];

  localStorage.setItem("dl_bgColor", bg);
  localStorage.setItem("dl_dmPref", dmPref);

  if (pfpFile) {
    const reader = new FileReader();
    reader.onload = e => {
      localStorage.setItem("dl_pfp", e.target.result);
      currentPfp = e.target.result;
      applySettings();
      closeSettings();
    };
    reader.readAsDataURL(pfpFile);
  } else {
    applySettings();
    closeSettings();
  }
}

function openSettings() {
  document.getElementById("settings-overlay").style.display = "flex";
}

function closeSettings() {
  document.getElementById("settings-overlay").style.display = "none";
}

function logout() {
  currentUser = null;
  currentPfp = null;
  localStorage.removeItem("dl_username");
  localStorage.removeItem("dl_password");
  document.getElementById("user-name").textContent = "User";
  document.getElementById("user-pfp").src = "";
  document.getElementById("login-overlay").style.display = "flex";
}

function restoreState() {
  const channels = JSON.parse(localStorage.getItem("dl_channels") || "[]");
  if (channels.length === 0) {
    channelList = ["general", "gaming", "random"];
  } else {
    channelList = channels;
  }
  renderChannels();

  const friends = JSON.parse(localStorage.getItem("dl_friends") || "[]");
  friendList = friends;
  renderFriends();

  listenInvites();
}

function renderChannels() {
  const listEl = document.getElementById("channels-list");
  listEl.innerHTML = "";
  channelList.forEach(name => {
    const div = document.createElement("div");
    div.className = "channel";
    div.setAttribute("data-channel", name);
    div.textContent = "#" + name;
    div.onclick = () => {
      document.querySelectorAll(".channel").forEach(c => c.classList.remove("active-channel"));
      document.querySelectorAll(".friend").forEach(f => f.classList.remove("active-friend"));
      div.classList.add("active-channel");
      currentChannel = name;
      document.getElementById("header").textContent = "#" + currentChannel;
      localStorage.setItem("dl_lastChannel", currentChannel);
      loadMessages(currentChannel);
    };
    listEl.appendChild(div);
  });
}

function setActiveChannel(name) {
  currentChannel = name;
  document.getElementById("header").textContent = "#" + currentChannel;
  document.querySelectorAll(".channel").forEach(c => {
    c.classList.remove("active-channel");
    if (c.getAttribute("data-channel") === name) c.classList.add("active-channel");
  });
  localStorage.setItem("dl_lastChannel", currentChannel);
}

function addChannel(name) {
  if (!name) return;
  if (channelList.includes(name)) return;
  channelList.push(name);
  localStorage.setItem("dl_channels", JSON.stringify(channelList));
  renderChannels();
}

function renderFriends() {
  const listEl = document.getElementById("friends-list");
  listEl.innerHTML = "";
  friendList.forEach(name => {
    const div = document.createElement("div");
    div.className = "friend";
    div.textContent = name;
    div.onclick = () => openDM(name);
    listEl.appendChild(div);
  });
}

function addFriend(name) {
  if (!name) return;
  if (friendList.includes(name)) return;
  friendList.push(name);
  localStorage.setItem("dl_friends", JSON.stringify(friendList));
  renderFriends();
}

// DM SYSTEM
function dmChannelName(a, b) {
  const sorted = [a, b].sort();
  return "dm_" + sorted[0] + "_" + sorted[1];
}

function openDM(friendName) {
  const dm = dmChannelName(currentUser, friendName);
  currentChannel = dm;

  document.getElementById("header").textContent = "@" + friendName;

  document.querySelectorAll(".friend").forEach(f => f.classList.remove("active-friend"));
  [...document.getElementById("friends-list").children].forEach(el => {
    if (el.textContent === friendName) el.classList.add("active-friend");
  });

  document.querySelectorAll(".channel").forEach(c => c.classList.remove("active-channel"));

  loadMessages(dm);
}

// MESSAGE RENDERING
function appendMessage(msg) {
  const messages = document.getElementById("messages");

  const row = document.createElement("div");
  row.className = "message-row";

  const pfp = document.createElement("img");
  pfp.className = "message-pfp";
  pfp.src = msg.pfp || defaultPfp(msg.user.charAt(0).toUpperCase());

  const content = document.createElement("div");
  content.className = "message-content";

  const name = document.createElement("span");
  name.className = "message-username";
  name.textContent = msg.user + ": ";

  content.appendChild(name);

  if (msg.type === "text") {
    const t = document.createElement("span");
    t.className = "message-text";
    t.textContent = msg.text;
    content.appendChild(t);
  }

  if (msg.type === "image") {
    const img = document.createElement("img");
    img.className = "message-image";
    img.src = msg.data;
    img.onclick = () => openImageModal(msg.data);
    content.appendChild(img);
  }

  if (msg.type === "video") {
    const wrap = document.createElement("div");
    wrap.className = "message-video-thumb-wrapper";

    const thumb = document.createElement("img");
    thumb.className = "message-video-thumb";
    thumb.src = msg.thumbnail;

    const play = document.createElement("div");
    play.className = "message-video-play";
    play.textContent = "▶";

    wrap.appendChild(thumb);
    wrap.appendChild(play);

    wrap.onclick = () => openVideoModal(msg.data, msg.filename);

    content.appendChild(wrap);
  }

  if (msg.type === "file") {
    const card = document.createElement("div");
    card.className = "message-file-card";

    const nameEl = document.createElement("div");
    nameEl.className = "message-file-name";
    nameEl.textContent = "📄 " + msg.filename;

    const sizeEl = document.createElement("div");
    sizeEl.className = "message-file-size";
    sizeEl.textContent = (msg.size / 1024).toFixed(1) + " KB";

    const btn = document.createElement("button");
    btn.className = "message-file-download";
    btn.textContent = "Download";
    btn.onclick = () => downloadBase64(msg.data, msg.filename);

    card.appendChild(nameEl);
    card.appendChild(sizeEl);
    card.appendChild(btn);

    content.appendChild(card);
  }

  row.appendChild(pfp);
  row.appendChild(content);
  messages.appendChild(row);
  messages.scrollTop = messages.scrollHeight;
}

// LOAD MESSAGES
function loadMessages(channel) {
  const messages = document.getElementById("messages");
  messages.innerHTML = "";

  db.ref("messages/" + channel).off();

  db.ref("messages/" + channel).on("value", snap => {
    messages.innerHTML = "";
    const data = snap.val();
    if (!data) return;

    Object.values(data).forEach(msg => appendMessage(msg));
  });
}

// SEND TEXT
function sendTextMessage() {
  const input = document.getElementById("message-input");
  const text = input.value.trim();
  if (!text) return;

  const msg = {
    user: currentUser,
    pfp: currentPfp || null,
    type: "text",
    text: text,
    ts: Date.now()
  };

  db.ref("messages/" + currentChannel).push(msg);
  input.value = "";
}

// FILE UPLOAD
document.getElementById("upload-button").onclick = () => {
  document.getElementById("file-input").click();
};

document.getElementById("file-input").onchange = async e => {
  const files = [...e.target.files];
  for (let file of files) {
    await handleFileUpload(file);
  }
  e.target.value = "";
};

async function handleFileUpload(file) {
  const reader = new FileReader();

  reader.onload = async e => {
    const base64 = e.target.result;

    if (file.type.startsWith("image/")) {
      sendImage(base64);
      return;
    }

    if (file.type.startsWith("video/")) {
      const thumb = await generateVideoThumbnail(file);
      sendVideo(base64, file.name, thumb);
      return;
    }

    sendFile(base64, file.name, file.size);
  };

  reader.readAsDataURL(file);
}

// VIDEO THUMBNAIL
function generateVideoThumbnail(file) {
  return new Promise(resolve => {
    const video = document.createElement("video");
    video.src = URL.createObjectURL(file);
    video.currentTime = 0.1;

    video.onloadeddata = () => {
      const canvas = document.createElement("canvas");
      canvas.width = 160;
      canvas.height = 90;
      const ctx = canvas.getContext("2d");
      ctx.drawImage(video, 0, 0, 160, 90);
      resolve(canvas.toDataURL("image/jpeg"));
    };
  });
}

// INVITES
function inviteFriend(name) {
  if (!name) return;
  db.ref("invites/" + name + "/" + currentUser).set({ ts: Date.now() });
}

function listenInvites() {
  if (!currentUser) return;
  db.ref("invites/" + currentUser).off();
  db.ref("invites/" + currentUser).on("value", snap => {
    const data = snap.val() || {};
    const dmPref = localStorage.getItem("dl_dmPref") || "on";
    const listEl = document.getElementById("invites-list");
    listEl.innerHTML = "";
    Object.keys(data).forEach(from => {
      if (dmPref === "on") {
        addFriend(from);
        db.ref("invites/" + currentUser + "/" + from).remove();
        return;
      }
      const row = document.createElement("div");
      row.className = "friend";
      row.textContent = "Invite from " + from;
      const accept = document.createElement("button");
      accept.textContent = "Accept";
      accept.style.marginLeft = "6px";
      accept.onclick = () => {
        addFriend(from);
        db.ref("invites/" + currentUser + "/" + from).remove();
      };
      const decline = document.createElement("button");
      decline.textContent = "Decline";
      decline.style.marginLeft = "4px";
      decline.onclick = () => {
        db.ref("invites/" + currentUser + "/" + from).remove();
      };
      const container = document.createElement("div");
      container.style.display = "flex";
      container.style.alignItems = "center";
      container.style.gap = "4px";
      container.appendChild(row);
      container.appendChild(accept);
      container.appendChild(decline);
      listEl.appendChild(container);
    });
  });
}
