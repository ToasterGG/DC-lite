// ─────────────────────────────────────────────
//  Account ID  (15-digit, unique per username+password pair)
//  Path: /profiles/<username>/id  →  "123456789012345"
//
//  The ID is derived from a hash of "username:password" so:
//   - Two users with the same name but different passwords get different IDs
//   - The same user always gets the same ID on any device (deterministic)
//   - The password is never stored in Firebase — only the derived ID is

// Simple but effective 32-bit hash (variant of FNV-1a)
function hashString(str) {
  let h = 2166136261;
  for (let i = 0; i < str.length; i++) {
    h ^= str.charCodeAt(i);
    h = (h * 16777619) >>> 0;  // keep 32-bit unsigned
  }
  return h;
}

// Seeded PRNG (mulberry32) — gives deterministic sequence from one seed
function mulberry32(seed) {
  return function() {
    seed |= 0; seed = seed + 0x6D2B79F5 | 0;
    let t = Math.imul(seed ^ seed >>> 15, 1 | seed);
    t = t + Math.imul(t ^ t >>> 7, 61 | t) ^ t;
    return ((t ^ t >>> 14) >>> 0) / 4294967296;
  };
}

function deriveIdFromCredentials(username, password) {
  // Hash the combined credential string to a 32-bit seed
  const seed = hashString(username + ":" + password);
  const rand  = mulberry32(seed);
  // Generate 15 digits: first must be 1-9
  let id = String(Math.floor(rand() * 9) + 1);
  for (let i = 0; i < 14; i++) id += String(Math.floor(rand() * 10));
  return id;
}

// ─────────────────────────────────────────────
//  Device fingerprint  (used for blocking)
//  A lightweight fingerprint stored in localStorage under a fixed key.
//  On block, admin writes it to /blocked/<fp> = true.
//  On every login, the client checks /blocked/<fp> before proceeding.
// ─────────────────────────────────────────────
function getDeviceFingerprint() {
  let fp = lsGet("dl_device_fp");
  if (fp) return fp;
  // Generate once and persist — combination of random + browser traits
  const traits = [
    navigator.userAgent,
    navigator.language,
    screen.width + "x" + screen.height,
    new Date().getTimezoneOffset(),
    Math.random().toString(36).slice(2)
  ].join("|");
  let h = 2166136261;
  for (let i = 0; i < traits.length; i++) {
    h ^= traits.charCodeAt(i);
    h = (h * 16777619) >>> 0;
  }
  fp = "fp_" + h.toString(16).padStart(8, "0");
  lsSet("dl_device_fp", fp);
  return fp;
}

function checkDeviceBlocked(fp, callback) {
  db.ref("blocked/" + fp).once("value", snap => {
    callback(!!snap.val());
  });
}

function friendsKey(u) { return "dl_friends_" + u; }

function loadFriends(username) {
  try { return JSON.parse(lsGet(friendsKey(username))) || []; }
  catch(_) { return []; }
}

function persistFriend(username, friendName) {
  const list = loadFriends(username);
  if (list.includes(friendName)) return false;
  list.push(friendName);
  lsSet(friendsKey(username), JSON.stringify(list));
  return true;
}

function removeFriend(username, friendName) {
  const list = loadFriends(username).filter(n => n !== friendName);
  lsSet(friendsKey(username), JSON.stringify(list));
}

function setBadge(channelKey, count) {
  const el = document.querySelector(`[data-channel-key="${channelKey}"]`);
  if (!el) return;
  let badge = el.querySelector(".unread-badge");
  if (count > 0) {
    if (!badge) {
      badge = document.createElement("span");
      badge.className = "unread-badge";
      el.appendChild(badge);
    }
    badge.textContent = count > 99 ? "99+" : count;
  } else {
    if (badge) badge.remove();
  }
}

function lastSeenKey(channelKey) {
  return "dl_seen_" + currentUser + "_" + channelKey;
}

function markChannelRead(channelKey) {
  lsSet(lastSeenKey(channelKey), Date.now().toString());
  if (window._badgeCounts) window._badgeCounts[channelKey] = 0;
  setBadge(channelKey, 0);
}

// Watch a channel and update its unread badge in real-time.
// Only counts messages from OTHER users that arrived after last seen.
function watchUnreads(channelKey) {
  if (watchingChannels[channelKey]) return;
  watchingChannels[channelKey] = true;

  // Use child_added for badge watching too — avoids full re-download
  // We track a local count per channel rather than scanning all messages
  const badgeCounts = badgeCounts || {};
  if (!window._badgeCounts) window._badgeCounts = {};

  db.ref("messages/" + channelKey).orderByChild("ts").on("child_added", snap => {
    if (channelKey === currentChannel) { markChannelRead(channelKey); return; }
    const msg = snap.val();
    if (!msg) return;
    const lastSeen = parseInt(lsGet(lastSeenKey(channelKey)) || "0", 10);
    if (msg.ts > lastSeen && msg.user !== currentUser) {
      window._badgeCounts[channelKey] = (window._badgeCounts[channelKey] || 0) + 1;
      setBadge(channelKey, window._badgeCounts[channelKey]);
    }
  });
}

// ─────────────────────────────────────────────
//  Login
// ─────────────────────────────────────────────
function handleLogin() {
  const u = document.getElementById("login-username").value.trim();
  const p = document.getElementById("login-password").value.trim();
  if (!u) return;

  // ── Pre-login checks ────────────────────────────────────────────────────
  const fp = getDeviceFingerprint();

  // 1. Check device block
  checkDeviceBlocked(fp, blocked => {
    if (blocked) {
      document.getElementById("login-username").value = "";
      document.getElementById("login-password").value = "";
      alert("This device has been blocked from Discord Lite+.");
      return;
    }

    // 2. Check account ban
    db.ref("banned/" + u).once("value", banSnap => {
      if (banSnap.val()) {
        alert("This account has been banned.");
        document.getElementById("login-username").value = "";
        document.getElementById("login-password").value = "";
        return;
      }

      // 3. Check mute status (load it so sendTextMessage can honour it)
      db.ref("muted/" + u).once("value", muteSnap => {
        window._isMuted = !!muteSnap.val();
      });

      // ── All clear — proceed with login ────────────────────────────────
      currentUser     = u;
      currentPassword = p;
      lsSet("dl_username", u);
      lsSet("dl_password", p);
      currentPfp = lsGet("dl_pfp_" + u) || null;
      if (currentPfp) profileCache[u] = currentPfp;

      // Detect admin
      isAdmin = (u === ADMIN_USER && p === ADMIN_PASS);

      // Record device fingerprint on the user's profile (admin can retrieve it)
      db.ref("profiles/" + u + "/deviceFp").set(fp);

      document.getElementById("login-overlay").style.display = "none";
      applyUserToUI();
      applySettings();
      loadOrCreateId(u, p);
      renderAllFriends(u);

      ["general", "gaming", "random"].forEach(ch => watchUnreads(ch));
      listenForDMInvites();
      runCleanup();
      loadMessages(currentChannel);
    });
  });
}

function dmChannelName(a, b) {
  return "dm_" + [a, b].sort().join("_");
}

// ── Cross-user DM invites ─────────────────────────────────────────────────
// When user A adds user B, we write:
//   /dmInvites/<userB>/<channelKey> → { from: "A", ts: ... }
// User B's client listens for child_added on /dmInvites/<userB>/
// and auto-adds A to their friends list.

function sendDMInvite(toUser) {
  const channelKey = dmChannelName(currentUser, toUser);
  db.ref("dmInvites/" + toUser + "/" + channelKey).set({
    from: currentUser,
    ts: Date.now()
  });
}

function listenForDMInvites() {
  db.ref("dmInvites/" + currentUser).on("child_added", snap => {
    const data = snap.val();
    if (!data) return;
    const fromUser = data.from;
    const added = persistFriend(currentUser, fromUser);
    if (added) renderFriendEntry(fromUser);
    snap.ref.remove(); // consume the invite so it doesn't re-fire on reload
  });
}

// ─────────────────────────────────────────────
//  AUTO-CLEANUP  (runs once on every login)
//
//  Three passes, all permanent deletes — no backups:
//    1. Messages  older than 24 h  →  deleted from every channel
//    2. DM channels with no message in 48 h  →  entire /messages/dm_… node deleted
//    3. User-presence records not touched in 72 h  →  /presence/<user> deleted
//
//  User presence is a tiny heartbeat record:
//    /presence/<username>  →  { ts: <epoch ms> }
//  We write/refresh it on login and every 10 minutes while the tab is open.
// ─────────────────────────────────────────────

const CLEANUP_MSG_TTL      = 24 * 60 * 60 * 1000;   // 24 h  — messages
const CLEANUP_DM_TTL       = 48 * 60 * 60 * 1000;   // 48 h  — idle DM channels
const CLEANUP_USER_TTL     = 72 * 60 * 60 * 1000;   // 72 h  — stale user records
const STATIC_CHANNELS      = ["general", "gaming", "random"];

// ── 1. Heartbeat ─────────────────────────────────────────────────────────────
function touchPresence() {
  if (!currentUser) return;
  db.ref("presence/" + currentUser).set({ ts: Date.now() });
}

// ── 2. Delete messages older than 24 h inside one channel ────────────────────
function purgeOldMessagesInChannel(channel) {
  const cutoff = Date.now() - CLEANUP_MSG_TTL;
  db.ref("messages/" + channel).once("value", snap => {
    const data = snap.val();
    if (!data) return;
    const updates = {};
    Object.entries(data).forEach(([key, msg]) => {
      if ((msg.ts || 0) < cutoff) updates[key] = null;
    });
    if (Object.keys(updates).length) {
      db.ref("messages/" + channel).update(updates);
    }
  });
}

// ── 3. Purge idle DM channels (last message > 48 h ago) ──────────────────────
function purgeIdleDMChannels() {
  const cutoff = Date.now() - CLEANUP_DM_TTL;
  db.ref("messages").once("value", snap => {
    const data = snap.val();
    if (!data) return;
    Object.entries(data).forEach(([channel, msgs]) => {
      if (!channel.startsWith("dm_")) return;
      const messages = Object.values(msgs || {});
      if (!messages.length) { db.ref("messages/" + channel).remove(); return; }
      const latest = Math.max(...messages.map(m => m.ts || 0));
      if (latest < cutoff) {
        db.ref("messages/" + channel).remove();
      }
    });
  });
}

// ── 4. Purge stale user presence records (last seen > 72 h ago) ──────────────
function purgeStaleUsers() {
  const cutoff = Date.now() - CLEANUP_USER_TTL;
  db.ref("presence").once("value", snap => {
    const data = snap.val();
    if (!data) return;
    Object.entries(data).forEach(([user, record]) => {
      if ((record.ts || 0) < cutoff) {
        db.ref("presence/" + user).remove();
        // Also remove any dmInvites queued for this user — they'll never log in
        db.ref("dmInvites/" + user).remove();
      }
    });
  });
}

// ── 5. Run all cleanups on login ─────────────────────────────────────────────
function runCleanup() {
  // Refresh this user's heartbeat immediately
  touchPresence();
  // Keep refreshing every 10 minutes while tab stays open
  setInterval(touchPresence, 10 * 60 * 1000);

  // Purge old messages from every static channel
  STATIC_CHANNELS.forEach(purgeOldMessagesInChannel);

  // Purge old messages from every DM channel this user is part of
  loadFriends(currentUser).forEach(friend => {
    purgeOldMessagesInChannel(dmChannelName(currentUser, friend));
  });

  // Purge whole idle DM channels and stale user records
  purgeIdleDMChannels();
  purgeStaleUsers();
}

// ─────────────────────────────────────────────
//  @mention rendering helpers
// ─────────────────────────────────────────────
function renderTextWithMentions(text) {
  const frag  = document.createDocumentFragment();
  const parts = text.split(/(@\w+)/g);
  parts.forEach(part => {
    if (part.startsWith("@")) {
      const span = document.createElement("span");
      span.className = "mention-highlight";
      span.textContent = part;
      frag.appendChild(span);
    } else {
      frag.appendChild(document.createTextNode(part));
    }
  });
  return frag;
}

function isMentionedInText(text) {
  if (!currentUser) return false;
  return new RegExp("@" + currentUser + "\\b", "i").test(text);
}

// ─────────────────────────────────────────────
//  Profile PFP cache
//  Fetches /profiles/<user> once per session, stores locally.
//  This means PFP data is fetched once, not embedded in every message.
// ─────────────────────────────────────────────
const profileCache = {}; // username → pfp data URL or "none"

function getProfilePfp(username) {
  if (!username) return defaultPfp("?");
  if (profileCache[username]) {
    return profileCache[username] === "none"
      ? defaultPfp(username.charAt(0).toUpperCase())
      : profileCache[username];
  }
  // Return default immediately; update DOM img elements when loaded
  if (profileCache[username] === undefined) {
    profileCache[username] = "pending";
    db.ref("profiles/" + username + "/pfp").once("value", snap => {
      const val = snap.val();
      profileCache[username] = val || "none";
      // Update any already-rendered avatars for this user
      document.querySelectorAll(`.message-pfp[data-user="${username}"]`).forEach(el => {
        el.src = val ? val : defaultPfp(username.charAt(0).toUpperCase());
      });
    });
  }
  return defaultPfp(username.charAt(0).toUpperCase());
}

function saveProfilePfp(pfpDataUrl) {
  if (!currentUser) return;
  if (pfpDataUrl) {
    db.ref("profiles/" + currentUser + "/pfp").set(pfpDataUrl);
    profileCache[currentUser] = pfpDataUrl;
  }
}

// ─────────────────────────────────────────────
//  Message rendering
// ─────────────────────────────────────────────
function appendMessage(msg, msgKey, msgChannel) {
  const messages = document.getElementById("messages");
  const row = document.createElement("div");
  row.className = "message-row";
  if (msgKey)     row.setAttribute("data-msg-key", msgKey);
  if (msgChannel) row.setAttribute("data-msg-channel", msgChannel);

  if (msg.type === "text" && isMentionedInText(msg.text)) {
    row.classList.add("mentioned");
  }

  const pfp = document.createElement("img");
  pfp.className = "message-pfp";
  pfp.setAttribute("data-user", msg.user);
  pfp.src = getProfilePfp(msg.user);

  const content = document.createElement("div");
  content.className = "message-content";

  const name = document.createElement("span");
  name.className = "message-username has-id-tooltip";
  // Crown badge for admin's own messages
  if (msg.user === ADMIN_USER) {
    const crown = document.createElement("span");
    crown.className = "admin-crown";
    crown.textContent = "👑";
    name.appendChild(crown);
    name.appendChild(document.createTextNode(" "));
  }
  name.appendChild(document.createTextNode(msg.user + ": "));
  // Tooltip element — populated lazily on first hover
  const tooltip = document.createElement("span");
  tooltip.className = "id-tooltip";
  tooltip.textContent = "loading…";
  name.appendChild(tooltip);
  name.addEventListener("mouseenter", () => {
    fetchUserId(msg.user, id => { tooltip.textContent = "ID: " + id; });
  }, { once: true });
  // Admin: click username to open admin panel for this user/message
  if (isAdmin && msg.user !== ADMIN_USER) {
    name.style.cursor = "pointer";
    name.addEventListener("click", e => {
      e.stopPropagation();
      const key = row.getAttribute("data-msg-key");
      const ch  = row.getAttribute("data-msg-channel");
      // Fetch target's device fingerprint
      db.ref("profiles/" + msg.user + "/deviceFp").once("value", snap => {
        openAdminPanel(msg.user, key, ch, snap.val());
      });
    });
  }
  content.appendChild(name);

  if (msg.type === "text") {
    const wrapper = document.createElement("span");
    wrapper.className = "message-text";
    wrapper.appendChild(renderTextWithMentions(msg.text));
    content.appendChild(wrapper);
  }

  if (msg.type === "image") {
    const local = mediaStore[msg.token];
    if (local) {
      const img = document.createElement("img");
      img.className = "message-image";
      img.src = local.url;
      img.onclick = () => openImageModal(local.url);
      content.appendChild(img);
    } else {
      // Sent from another session — we don't have the bytes
      const note = document.createElement("span");
      note.className = "message-text";
      note.style.color = "#72767d";
      note.textContent = "🖼️ " + (msg.filename || "image") + " (shared from another session)";
      content.appendChild(note);
    }
  }

  if (msg.type === "video") {
    const local = mediaStore[msg.token];
    const wrap  = document.createElement("div");
    wrap.className = "message-video-thumb-wrapper";
    if (local && msg.thumbnail) {
      const thumb = document.createElement("img");
      thumb.className = "message-video-thumb";
      thumb.src = msg.thumbnail;
      const play = document.createElement("div");
      play.className = "message-video-play";
      play.textContent = "▶";
      wrap.appendChild(thumb);
      wrap.appendChild(play);
      wrap.onclick = () => openVideoModal(local.url, msg.filename);
      content.appendChild(wrap);
    } else {
      const note = document.createElement("span");
      note.className = "message-text";
      note.style.color = "#72767d";
      note.textContent = "🎬 " + (msg.filename || "video") + " (shared from another session)";
      content.appendChild(note);
    }
  }

  if (msg.type === "file") {
    const local = mediaStore[msg.token];
    const card  = document.createElement("div");
    card.className = "message-file-card";
    const nameEl = document.createElement("div");
    nameEl.className = "message-file-name";
    nameEl.textContent = "📄 " + (msg.filename || "file");
    card.appendChild(nameEl);
    if (msg.size) {
      const sizeEl = document.createElement("div");
      sizeEl.className = "message-file-size";
      sizeEl.textContent = (msg.size / 1024).toFixed(1) + " KB";
      card.appendChild(sizeEl);
    }
    if (local) {
      const btn = document.createElement("button");
      btn.className = "message-file-download";
      btn.textContent = "Download";
      btn.onclick = () => {
        const a = document.createElement("a");
        a.href = local.url; a.download = msg.filename; a.click();
      };
      card.appendChild(btn);
    } else {
      const note = document.createElement("div");
      note.style.cssText = "font-size:11px;color:#72767d;margin-top:2px;";
      note.textContent = "Shared from another session";
      card.appendChild(note);
    }
    content.appendChild(card);
  }

  row.appendChild(pfp);
  row.appendChild(content);
  messages.appendChild(row);
  messages.scrollTop = messages.scrollHeight;
}

// ─────────────────────────────────────────────
//  Load messages
//
//  BANDWIDTH FIX: use child_added instead of on("value").
//  on("value") re-downloads the ENTIRE channel on every new message.
//  child_added fires once per existing message on attach, then
//  once per new message only — never re-downloads old ones.
// ─────────────────────────────────────────────
let currentChannelRef = null;

function loadMessages(channel) {
  const messages = document.getElementById("messages");
  messages.innerHTML = "";

  if (currentChannelRef) {
    currentChannelRef.off("child_added");
    currentChannelRef = null;
  }

  const ref = db.ref("messages/" + channel).orderByChild("ts");
  currentChannelRef = ref;

  ref.on("child_added", snap => {
    const msg = snap.val();
    if (msg) appendMessage(msg, snap.key, channel);
  });

  markChannelRead(channel);
}

//  Send text
// ─────────────────────────────────────────────
function sendTextMessage() {
  const input = document.getElementById("message-input");
  const text  = input.value.trim();
  if (!text) return;
  if (window._isMuted) {
    input.value = "";
    const tmp = input.placeholder;
    input.placeholder = "You are muted and cannot send messages.";
    setTimeout(() => { input.placeholder = tmp; }, 3000);
    return;
  }
  closeMentionDropdown();
  db.ref("messages/" + currentChannel).push({
    user: currentUser,
    type: "text", text, ts: Date.now()
  });
  input.value = "";
}

// ─────────────────────────────────────────────
//  @mention autocomplete
// ─────────────────────────────────────────────
let mentionDropdownActive = false;
let mentionDropdownIndex  = -1;
let mentionStart          = -1;

function getMentionCandidates(query) {
  return loadFriends(currentUser).filter(
    n => n.toLowerCase().startsWith(query.toLowerCase())
  );
}

function openMentionDropdown(candidates) {
  const dd = document.getElementById("mention-dropdown");
  dd.innerHTML = "";
  if (!candidates.length) { closeMentionDropdown(); return; }

  candidates.forEach(name => {
    const opt = document.createElement("div");
    opt.className = "mention-option";
    opt.textContent = "@" + name;
    opt.onmousedown = e => { e.preventDefault(); selectMention(name); };
    dd.appendChild(opt);
  });

  dd.style.display = "block";
  mentionDropdownActive = true;
  mentionDropdownIndex  = -1;
}

function closeMentionDropdown() {
  document.getElementById("mention-dropdown").style.display = "none";
  mentionDropdownActive = false;
  mentionDropdownIndex  = -1;
}

function updateDropdownSelection() {
  document.querySelectorAll(".mention-option").forEach((o, i) =>
    o.classList.toggle("selected", i === mentionDropdownIndex)
  );
}

function selectMention(name) {
  const input = document.getElementById("message-input");
  const val   = input.value;
  const after = val.slice(input.selectionEnd);
  input.value = val.slice(0, mentionStart) + "@" + name + " " + after;
  const pos = mentionStart + name.length + 2;
  input.setSelectionRange(pos, pos);
  closeMentionDropdown();
  input.focus();
}

const msgInput = document.getElementById("message-input");

msgInput.addEventListener("input", function() {
  const val   = this.value;
  const caret = this.selectionStart;
  // Walk left to find start of current word
  let i = caret - 1;
  while (i >= 0 && val[i] !== " " && val[i] !== "\n") i--;
  const wordStart = i + 1;
  const word = val.slice(wordStart, caret);

  if (word.startsWith("@")) {
    mentionStart = wordStart;
    openMentionDropdown(getMentionCandidates(word.slice(1)));
  } else {
    closeMentionDropdown();
  }
});

msgInput.addEventListener("keydown", function(e) {
  if (!mentionDropdownActive) {
    if (e.key === "Enter") { e.preventDefault(); sendTextMessage(); }
    return;
  }
  const opts = document.querySelectorAll(".mention-option");
  if (e.key === "ArrowDown") {
    e.preventDefault();
    mentionDropdownIndex = Math.min(mentionDropdownIndex + 1, opts.length - 1);
    updateDropdownSelection();
  } else if (e.key === "ArrowUp") {
    e.preventDefault();
    mentionDropdownIndex = Math.max(mentionDropdownIndex - 1, 0);
    updateDropdownSelection();
  } else if (e.key === "Tab" || e.key === "Enter") {
    e.preventDefault();
    if (mentionDropdownIndex >= 0 && opts[mentionDropdownIndex]) {
      selectMention(opts[mentionDropdownIndex].textContent.slice(1));
    } else if (e.key === "Enter") {
      closeMentionDropdown();
      sendTextMessage();
    }
  } else if (e.key === "Escape") {
    closeMentionDropdown();
  }
});

document.addEventListener("click", e => {
  if (!document.getElementById("mention-dropdown").contains(e.target) &&
      e.target.id !== "message-input") {
    closeMentionDropdown();
  }
});

// ── BANDWIDTH FIX: media/files are NEVER stored in Firebase ──────────────────
// Base64 in the database was the primary cause of massive download costs.
// Instead we keep a local mediaStore map: token → objectURL.
// Only a lightweight stub (type, filename, size, token) goes to Firebase.
// The actual bytes live in the sender's browser session only.
// Recipients see a "file shared — ask sender for the file" notice if they
// weren't in the same session. This is unavoidable without paid storage.
const mediaStore = {}; // token → { url, filename }

function makeToken() {
  return Math.random().toString(36).slice(2) + Date.now().toString(36);
}

async function handleFileUpload(file) {
  const url   = URL.createObjectURL(file);
  const token = makeToken();
  mediaStore[token] = { url, filename: file.name };

  if (file.type.startsWith("image/")) {
    sendImage(token, file.name);
  } else if (file.type.startsWith("video/")) {
    const thumb = await generateVideoThumbnail(file);
    sendVideo(token, file.name, thumb);
  } else {
    sendFile(token, file.name, file.size);
  }
}

function sendImage(token, filename) {
  db.ref("messages/" + currentChannel).push({
    user: currentUser, type: "image", token, filename, ts: Date.now()
  });
}
function sendVideo(token, filename, thumbnail) {
  // thumbnail is a tiny 160x90 jpeg — keep it, it's small (~3 KB)
  db.ref("messages/" + currentChannel).push({
    user: currentUser, type: "video", token, filename, thumbnail, ts: Date.now()
  });
}
function sendFile(token, filename, size) {
  db.ref("messages/" + currentChannel).push({
    user: currentUser, type: "file", token, filename, size, ts: Date.now()
  });
}

function generateVideoThumbnail(file) {
  return new Promise(resolve => {
    const video = document.createElement("video");
    video.src = URL.createObjectURL(file);
    video.currentTime = 0.1;
    video.onloadeddata = () => {
      const canvas = document.createElement("canvas");
      canvas.width = 160; canvas.height = 90;
      canvas.getContext("2d").drawImage(video, 0, 0, 160, 90);
      resolve(canvas.toDataURL("image/jpeg"));
    };
  });
}

function downloadBase64(data, filename) {
  const a = document.createElement("a");
  a.href = data; a.download = filename; a.click();
}
